var delay = 20

func Bardiche()
  equip bardiche 
  activate r


func ices()
  equipL ice wand +21
  equipR ice wand +16
  

func BFG()
  ?item.GetCooldown("blade")=0 &
  ^foe.count > 5 &
  ^foe.distance < 30
    screen.next()
    delay--
    ?delay=0
      equip blade
      activate r
      screen.ResetOffset()
      delay = 20


func QuarterUse()
  ?item.GetCooldown("quarterstaff")<1 &
  ^item.CanActivate()
    equip quarterstaff *10
    activate r


func RuneShield()
  ?foe = poison
    equipR ice shield *10 A
  :?foe = vigor
    equipR poison shield *10 A
  :?foe = aether
    equipR vigor shield A *10
  :?foe = fire
    equipR aether shield A *10
  :?foe = ice
    equipR fire shield A *10
  :
    equipR ice shield A *10
  equipL triskelion


func Compound()
  equipR compound *10
  equipL triskelion

func Dashing()
  ?item.GetCooldown("bash") <= 0
    ?item.GetCooldown("dash") <= 0
      ?foe.count >= 4
        equipR bashing shield
        equipL triskelion
      :?foe = boss & foe ! Poena
        equipR bashing shield
        equipL triskelion
      :
        equipR dashing shield
        equipL triskelion
    :
      ?foe ! Poena | (loc = Caves & foe = Ceiling Decorator)
        equipR bashing shield
        equipL triskelion
      :
        equipR dashing shield
        equipL triskelion
  :
    ?item.GetCooldown("dash") <= 0
      equipR dashing shield
      equipL triskelion


func CanPickup()
  equipR star stone
  equipL triskelion
